44      private double getValue(String json, String timerName, String timerSection,String value, HttpMetricType type)
45      {
46          Map m = new GsonBuilder().create().fromJson(json,Map.class);
47          Map<String,Map> requests =  getRequestOrResponseMap(m,type.toString()).get(timerName);
48          Double val = (Double)requests.get(timerSection).get(value);
