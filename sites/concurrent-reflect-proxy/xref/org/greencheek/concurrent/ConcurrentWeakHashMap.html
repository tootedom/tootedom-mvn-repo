31  /**
32   * A hash table with <em>weak keys</em>, full concurrency of retrievals, and
33   * adjustable expected concurrency for updates. Similar to
34   * {@link java.util.WeakHashMap}, entries of this table are periodically
35   * removed once their corresponding keys are no longer referenced outside of
36   * this table. In other words, this table will not prevent a key from being
37   * discarded by the garbage collector. Once a key has been discarded by the
38   * collector, the corresponding entry is no longer visible to this table;
39   * however, the entry may occupy space until a future table operation decides to
40   * reclaim it. For this reason, summary functions such as <tt>size</tt> and
41   * <tt>isEmpty</tt> might return a value greater than the observed number of
42   * entries. In order to support a high level of concurrency, stale entries are
43   * only reclaimed during blocking (usually mutating) operations.
44   * 
45   * While keys in this table are only held using a weak reference, values are
46   * held using a normal strong reference. This provides the guarantee that a
47   * value will always have at least the same life-span as it's key. For this
48   * reason, care should be taken to ensure that a value never refers, either
49   * directly or indirectly, to its key, thereby preventing reclamation. If weak
50   * values are desired, one can simply use a {@link WeakReference} for the value
51   * type.
52   * 
53   * Just like {@link java.util.ConcurrentHashMap}, this class obeys the same
54   * functional specification as {@link java.util.Hashtable}, and includes
55   * versions of methods corresponding to each method of <tt>Hashtable</tt>.
56   * However, even though all operations are thread-safe, retrieval operations do
57   * <em>not</em> entail locking, and there is <em>not</em> any support for
58   * locking the entire table in a way that prevents all access. This class is
59   * fully interoperable with <tt>Hashtable</tt> in programs that rely on its
60   * thread safety but not on its synchronization details.
61   * 
62   * <p>
63   * Retrieval operations (including <tt>get</tt>) generally do not block, so
64   * may overlap with update operations (including <tt>put</tt> and
65   * <tt>remove</tt>). Retrievals reflect the results of the most recently
66   * <em>completed</em> update operations holding upon their onset. For
67   * aggregate operations such as <tt>putAll</tt> and <tt>clear</tt>,
68   * concurrent retrievals may reflect insertion or removal of only some entries.
69   * Similarly, Iterators and Enumerations return elements reflecting the state of
70   * the hash table at some point at or since the creation of the
71   * iterator/enumeration. They do <em>not</em> throw
72   * {@link ConcurrentModificationException}. However, iterators are designed to
73   * be used by only one thread at a time.
74   * 
75   * <p>
76   * The allowed concurrency among update operations is guided by the optional
77   * <tt>concurrencyLevel</tt> constructor argument (default <tt>16</tt>),
78   * which is used as a hint for internal sizing. The table is internally
79   * partitioned to try to permit the indicated number of concurrent updates
80   * without contention. Because placement in hash tables is essentially random,
81   * the actual concurrency will vary. Ideally, you should choose a value to
82   * accommodate as many threads as will ever concurrently modify the table. Using
83   * a significantly higher value than you need can waste space and time, and a
84   * significantly lower value can lead to thread contention. But overestimates
85   * and underestimates within an order of magnitude do not usually have much
86   * noticeable impact. A value of one is appropriate when it is known that only
87   * one thread will modify and all others will only read. Also, resizing this or
88   * any other kind of hash table is a relatively slow operation, so, when
89   * possible, it is a good idea to provide estimates of expected table sizes in
90   * constructors.
91   * 
92   * <p>
93   * This class and its views and iterators implement all of the <em>optional</em>
94   * methods of the {@link Map} and {@link Iterator} interfaces.
95   * 
96   * <p>
97   * Like {@link Hashtable} but unlike {@link HashMap}, this class does
98   * <em>not</em> allow <tt>null</tt> to be used as a key or value.
99   * 
100  * <p>
101  * This class is a member of the <a href="{@docRoot}/../technotes/guides/collections/index.html">
102  * Java Collections Framework</a>.
103  * 
104  * @author Doug Lea
105  * @author Jason T. Greene
106  * @param <K> the type of keys maintained by this map
107  * @param <V> the type of mapped values
108  */
109 public class ConcurrentWeakHashMap<K, V> extends AbstractMap<K, V>
110         implements java.util.concurrent.ConcurrentMap<K, V>, Serializable {

184     /**
185      * Applies a supplemental hash function to a given hashCode, which
186      * defends against poor quality hash functions.  This is critical
187      * because ConcurrentWeakHashMap uses power-of-two length hash tables,
188      * that otherwise encounter collisions for hashCodes that do not
189      * differ in lower or upper bits.
190      */
191     private static int hash(int h) {
192         // Spread bits to regularize both segment and index locations,
193         // using variant of single-word Wang/Jenkins hash.
194         h += (h <<  15) ^ 0xffffcd7d;
195         h ^= (h >>> 10);
196         h += (h <<   3);
197         h ^= (h >>>  6);
198         h += (h <<   2) + (h << 14);
199         return h ^ (h >>> 16);
200     }
201 
202     /**
203      * Returns the segment that should be used for key with given hash
204      * @param hash the hash code for the key
205      * @return the segment
206      */
207     final Segment<K,V> segmentFor(int hash) {
208         return segments[(hash >>> segmentShift) & segmentMask];
209     }

255     /**
256      * Segments are specialized versions of hash tables.  This
257      * subclasses from ReentrantLock opportunistically, just to
258      * simplify some locking and avoid separate construction.
259      */
260     static final class Segment<K,V> extends ReentrantLock implements Serializable {
261         /*
262          * Segments maintain a table of entry lists that are ALWAYS
263          * kept in a consistent state, so can be read without locking.
264          * Next fields of nodes are immutable (final).  All list
265          * additions are performed at the front of each bin. This
266          * makes it easy to check changes, and also fast to traverse.
267          * When nodes would otherwise be changed, new nodes are
268          * created to replace them. This works well for hash tables
269          * since the bin lists tend to be short. (The average length
270          * is less than two for the default load factor threshold.)
271          *
272          * Read operations can thus proceed without locking, but rely
273          * on selected uses of volatiles to ensure that completed
274          * write operations performed by other threads are
275          * noticed. For most purposes, the "count" field, tracking the
276          * number of elements, serves as that volatile variable
277          * ensuring visibility.  This is convenient because this field
278          * needs to be read in many read operations anyway:
279          *
280          *   - All (unsynchronized) read operations must first read the
281          *     "count" field, and should not look at table entries if
282          *     it is 0.
283          *
284          *   - All (synchronized) write operations should write to
285          *     the "count" field after structurally changing any bin.
286          *     The operations must not take any action that could even
287          *     momentarily cause a concurrent read operation to see
288          *     inconsistent data. This is made easier by the nature of
289          *     the read operations in Map. For example, no operation
290          *     can reveal that the table has grown but the threshold
291          *     has not yet been updated, so there are no atomicity
292          *     requirements for this with respect to reads.
293          *
294          * As a guide, all critical volatile reads and writes to the
295          * count field are marked in code comments.
296          */
297

377             lock();
378             try {

388         V get(Object key, int hash) {
389             if (count != 0) { // read-volatile
390                 HashEntry<K,V> e = getFirst(hash);
391                 while (e != null) {
392                     if (e.hash == hash && key.equals(e.keyRef.get())) {
393                         V v = e.value;
394                         if (v != null)
395                             return v;
396                         return readValueUnderLock(e); // recheck
397                     }
398                     e = e.next;
399                 }
400             }
401             return null;
402         }
403 
404         boolean containsKey(Object key, int hash) {
405             if (count != 0) { // read-volatile
406                 HashEntry<K,V> e = getFirst(hash);
407                 while (e != null) {
408                     if (e.hash == hash && key.equals(e.keyRef.get()))
409                         return true;
410                     e = e.next;
411                 }
412             }
413             return false;
414         }
415 
416         boolean containsValue(Object value) {
417             if (count != 0) { // read-volatile
418                 HashEntry<K,V>[] tab = table;
419                 int len = tab.length;
420                 for (int i = 0 ; i < len; i++) {
421                     for (HashEntry<K,V> e = tab[i]; e != null; e = e.next) {
422                         V v = e.value;
423                         if (v == null) // recheck
424                             v = readValueUnderLock(e);
425                         if (value.equals(v))
426                             return true;
427                     }
428                 }
429             }
430             return false;
431         }

433         boolean replace(K key, int hash, V oldValue, V newValue) {
434             lock();
435             try {
436                 removeStale();
437                 HashEntry<K,V> e = getFirst(hash);
438                 while (e != null && (e.hash != hash || !key.equals(e.keyRef.get())))
439                     e = e.next;
440 
441                 boolean replaced = false;
442                 if (e != null && oldValue.equals(e.value)) {
443                     replaced = true;
444                     e.value = newValue;
445                 }
446                 return replaced;
447             } finally {
448                 unlock();
449             }
450         }

472         V put(K key, int hash, V value, boolean onlyIfAbsent) {
473             lock();
474             try {
475                 removeStale();
476                 int c = count;
477                 if (c++ > threshold) {// ensure capacity
478                     int reduced = rehash();
479                     if (reduced > 0)  // adjust from possible weak cleanups
480                         count = (c -= reduced) - 1; // write-volatile      
481                 }
482                          
483                 HashEntry<K,V>[] tab = table;
484                 int index = hash & (tab.length - 1);
485                 HashEntry<K,V> first = tab[index];
486                 HashEntry<K,V> e = first;
487                 while (e != null && (e.hash != hash || !key.equals(e.keyRef.get())))
488                     e = e.next;
489 
490                 V oldValue;
491                 if (e != null) {
492                     oldValue = e.value;
493                     if (!onlyIfAbsent)
494                         e.value = value;
495                 }
496                 else {
497                     oldValue = null;
498                     ++modCount;
499                     tab[index] = new HashEntry<K,V>(key, hash, first, value, refQueue);
500                     count = c; // write-volatile
501                 }
502                 return oldValue;
503             } finally {
504                 unlock();
505             }
506         }

508         int rehash() {
509             HashEntry<K,V>[] oldTable = table;
510             int oldCapacity = oldTable.length;
511             if (oldCapacity >= MAXIMUM_CAPACITY)
512                 return 0;
513 
514             /*
515              * Reclassify nodes in each list to new Map.  Because we are
516              * using power-of-two expansion, the elements from each bin
517              * must either stay at same index, or move with a power of two
518              * offset. We eliminate unnecessary node creation by catching
519              * cases where old nodes can be reused because their next
520              * fields won't change. Statistically, at the default
521              * threshold, only about one-sixth of them need cloning when
522              * a table doubles. The nodes they replace will be garbage
523              * collectable as soon as they are no longer referenced by any
524              * reader thread that may be in the midst of traversing table
525              * right now.
526              */
527 
528             HashEntry<K,V>[] newTable = HashEntry.newArray(oldCapacity<<1);
529             threshold = (int)(newTable.length * loadFactor);
530             int sizeMask = newTable.length - 1;
531             int reduce = 0;
532             for (int i = 0; i < oldCapacity ; i++) {
533                 // We need to guarantee that any existing reads of old Map can
534                 //  proceed. So we cannot yet null out each bin.
535                 HashEntry<K,V> e = oldTable[i];
536 
537                 if (e != null) {
538                     HashEntry<K,V> next = e.next;
539                     int idx = e.hash & sizeMask;
540 
541                     //  Single node on list
542                     if (next == null)
543                         newTable[idx] = e;
544 
545                     else {
546                         // Reuse trailing consecutive sequence at same slot
547                         HashEntry<K,V> lastRun = e;
548                         int lastIdx = idx;
549                         for (HashEntry<K,V> last = next;
550                              last != null;
551                              last = last.next) {
552                             int k = last.hash & sizeMask;
553                             if (k != lastIdx) {
554                                 lastIdx = k;
555                                 lastRun = last;
556                             }
557                         }
558                         newTable[lastIdx] = lastRun;
559                         // Clone all remaining nodes
560                         for (HashEntry<K,V> p = e; p != lastRun; p = p.next) {
561                             // Skip GC'd weak refs
562                             K key = p.keyRef.get();
563                             if (key == null) {
564                                 reduce++;
565                                 continue;
566                             }
567                             int k = p.hash & sizeMask;
568                             HashEntry<K,V> n = newTable[k];
569                             newTable[k] = new HashEntry<K,V>(key, p.hash, n, p.value, refQueue);
570                         }
571                     }
572                 }
573             }
574             table = newTable;
575             return reduce;
576         }
577 
578         /**
579          * Remove; match on key only if value null, else match both.
580          */
581         V remove(Object key, int hash, Object value, boolean weakRemove) {
582             lock();
583             try {
584                 if (!weakRemove)
585                     removeStale();
586                 int c = count - 1;
587                 HashEntry<K,V>[] tab = table;
588                 int index = hash & (tab.length - 1);
589                 HashEntry<K,V> first = tab[index];
590                 HashEntry<K,V> e = first;
591                 // a weak remove operation compares the WeakReference instance
592                 while (e != null && (!weakRemove || key != e.keyRef) 
593                                  && (e.hash != hash || !key.equals(e.keyRef.get())))
594                     e = e.next;
595 
596                 V oldValue = null;
597                 if (e != null) {
598                     V v = e.value;
599                     if (value == null || value.equals(v)) {
600                         oldValue = v;
601                         // All entries following removed node can stay
602                         // in list, but all preceding ones need to be
603                         // cloned.
604                         ++modCount;
605                         HashEntry<K,V> newFirst = e.next;
606                         for (HashEntry<K,V> p = first; p != e; p = p.next) {
607                             K pKey = p.keyRef.get();
608                             if (pKey == null) { // Skip GC'd keys
609                                 c--;
610                                 continue;
611                             }
612                                 
613                             newFirst = new HashEntry<K,V>(pKey, p.hash,
614                                                           newFirst, p.value, refQueue);
615                         }
616                         tab[index] = newFirst;
617                         count = c; // write-volatile
618                     }
619                 }
620                 return oldValue;
621             } finally {
622                 unlock();
623             }
624         }
625         
626         @SuppressWarnings("unchecked")
627         void removeStale() {
628             WeakKeyReference<K> ref;
629             while ((ref = (WeakKeyReference<K>) refQueue.poll()) != null) {
630                 remove(ref, ref.hash, null, true);
631             }
632         }
633 
634         void clear() {
635             if (count != 0) {
636                 lock();
637                 try {
638                     HashEntry<K,V>[] tab = table;
639                     for (int i = 0; i < tab.length ; i++)
640                         tab[i] = null;
641                     ++modCount;
642                     // replace the reference queue to avoid unnecessary stale cleanups
643                     refQueue = new ReferenceQueue<K>();
644                     count = 0; // write-volatile
645                 } finally {
646                     unlock();
647                 }
648             }
649         }

656     /**
657      * Creates a new, empty map with the specified initial
658      * capacity, load factor and concurrency level.
659      *
660      * @param initialCapacity the initial capacity. The implementation
661      * performs internal sizing to accommodate this many elements.
662      * @param loadFactor  the load factor threshold, used to control resizing.
663      * Resizing may be performed when the average number of elements per
664      * bin exceeds this threshold.
665      * @param concurrencyLevel the estimated number of concurrently
666      * updating threads. The implementation performs internal sizing
667      * to try to accommodate this many threads.
668      * @throws IllegalArgumentException if the initial capacity is
669      * negative or the load factor or concurrencyLevel are
670      * nonpositive.
671      */
672     public ConcurrentWeakHashMap(int initialCapacity,
673                              float loadFactor, int concurrencyLevel) {
674         if (!(loadFactor > 0) || initialCapacity < 0 || concurrencyLevel <= 0)
675             throw new IllegalArgumentException();
676 
677         if (concurrencyLevel > MAX_SEGMENTS)
678             concurrencyLevel = MAX_SEGMENTS;
679 
680         // Find power-of-two sizes best matching arguments
681         int sshift = 0;
682         int ssize = 1;
683         while (ssize < concurrencyLevel) {
684             ++sshift;
685             ssize <<= 1;
686         }
687         segmentShift = 32 - sshift;
688         segmentMask = ssize - 1;
689         this.segments = Segment.newArray(ssize);
690 
691         if (initialCapacity > MAXIMUM_CAPACITY)
692             initialCapacity = MAXIMUM_CAPACITY;
693         int c = initialCapacity / ssize;
694         if (c * ssize < initialCapacity)
695             ++c;
696         int cap = 1;
697         while (cap < c)
698             cap <<= 1;
699 
700         for (int i = 0; i < this.segments.length; ++i)
701             this.segments[i] = new Segment<K,V>(cap, loadFactor);
702     }

758     /**
759      * Returns <tt>true</tt> if this map contains no key-value mappings.
760      *
761      * @return <tt>true</tt> if this map contains no key-value mappings
762      */
763     public boolean isEmpty() {
764         final Segment<K,V>[] segments = this.segments;
765         /*
766          * We keep track of per-segment modCounts to avoid ABA
767          * problems in which an element in one segment was added and
768          * in another removed during traversal, in which case the
769          * table was never actually empty at any point. Note the
770          * similar use of modCounts in the size() and containsValue()
771          * methods, which are the only other methods also susceptible
772          * to ABA problems.
773          */
774         int[] mc = new int[segments.length];
775         int mcsum = 0;
776         for (int i = 0; i < segments.length; ++i) {
777             if (segments[i].count != 0)
778                 return false;
779             else
780                 mcsum += mc[i] = segments[i].modCount;
781         }
782         // If mcsum happens to be zero, then we know we got a snapshot
783         // before any modifications at all were made.  This is
784         // probably common enough to bother tracking.
785         if (mcsum != 0) {
786             for (int i = 0; i < segments.length; ++i) {
787                 if (segments[i].count != 0 ||
788                     mc[i] != segments[i].modCount)
789                     return false;
790             }
791         }
792         return true;
793     }

795     /**
796      * Returns the number of key-value mappings in this map.  If the
797      * map contains more than <tt>Integer.MAX_VALUE</tt> elements, returns
798      * <tt>Integer.MAX_VALUE</tt>.
799      *
800      * @return the number of key-value mappings in this map
801      */
802     public int size() {
803         final Segment<K,V>[] segments = this.segments;
804         long sum = 0;
805         long check = 0;
806         int[] mc = new int[segments.length];
807         // Try a few times to get accurate count. On failure due to
808         // continuous async changes in table, resort to locking.
809         for (int k = 0; k < RETRIES_BEFORE_LOCK; ++k) {
810             check = 0;
811             sum = 0;
812             int mcsum = 0;
813             for (int i = 0; i < segments.length; ++i) {
814                 sum += segments[i].count;
815                 mcsum += mc[i] = segments[i].modCount;
816             }
817             if (mcsum != 0) {
818                 for (int i = 0; i < segments.length; ++i) {
819                     check += segments[i].count;
820                     if (mc[i] != segments[i].modCount) {
821                         check = -1; // force retry
822                         break;
823                     }
824                 }
825             }
826             if (check == sum)
827                 break;
828         }
829         if (check != sum) { // Resort to locking all segments
830             sum = 0;
831             for (int i = 0; i < segments.length; ++i)
832                 segments[i].lock();
833             for (int i = 0; i < segments.length; ++i)
834                 sum += segments[i].count;
835             for (int i = 0; i < segments.length; ++i)
836                 segments[i].unlock();
837         }
838         if (sum > Integer.MAX_VALUE)
839             return Integer.MAX_VALUE;
840         else
841             return (int)sum;
842     }

874     /**
875      * Returns <tt>true</tt> if this map maps one or more keys to the
876      * specified value. Note: This method requires a full internal
877      * traversal of the hash table, and so is much slower than
878      * method <tt>containsKey</tt>.
879      *
880      * @param value value whose presence in this map is to be tested
881      * @return <tt>true</tt> if this map maps one or more keys to the
882      *         specified value
883      * @throws NullPointerException if the specified value is null
884      */
885     public boolean containsValue(Object value) {
886         if (value == null)
887             throw new NullPointerException();
888 
889         // See explanation of modCount use above
890 
891         final Segment<K,V>[] segments = this.segments;
892         int[] mc = new int[segments.length];
893 
894         // Try a few times without locking
895         for (int k = 0; k < RETRIES_BEFORE_LOCK; ++k) {
896             int sum = 0;
897             int mcsum = 0;
898             for (int i = 0; i < segments.length; ++i) {
899                 int c = segments[i].count;
900                 mcsum += mc[i] = segments[i].modCount;
901                 if (segments[i].containsValue(value))
902                     return true;
903             }
904             boolean cleanSweep = true;
905             if (mcsum != 0) {
906                 for (int i = 0; i < segments.length; ++i) {
907                     int c = segments[i].count;
908                     if (mc[i] != segments[i].modCount) {
909                         cleanSweep = false;
910                         break;
911                     }
912                 }
913             }
914             if (cleanSweep)
915                 return false;
916         }
917         // Resort to locking all segments
918         for (int i = 0; i < segments.length; ++i)
919             segments[i].lock();
920         boolean found = false;
921         try {
922             for (int i = 0; i < segments.length; ++i) {
923                 if (segments[i].containsValue(value)) {
924                     found = true;
925                     break;
926                 }
927             }
928         } finally {
929             for (int i = 0; i < segments.length; ++i)
930                 segments[i].unlock();
931         }
932         return found;
933     }

1145     abstract class HashIterator {
1146         int nextSegmentIndex;
1147         int nextTableIndex;
1148         HashEntry<K,V>[] currentTable;
1149         HashEntry<K, V> nextEntry;
1150         HashEntry<K, V> lastReturned;
1151         K currentKey; // Strong reference to weak key (prevents gc)
1152 
1153         HashIterator() {
1154             nextSegmentIndex = segments.length - 1;
1155             nextTableIndex = -1;
1156             advance();
1157         }
1158 
1159         public boolean hasMoreElements() { return hasNext(); }
1160 
1161         final void advance() {
1162             if (nextEntry != null && (nextEntry = nextEntry.next) != null)
1163                 return;
1164 
1165             while (nextTableIndex >= 0) {
1166                 if ( (nextEntry = currentTable[nextTableIndex--]) != null)
1167                     return;
1168             }
1169 
1170             while (nextSegmentIndex >= 0) {
1171                 Segment<K,V> seg = segments[nextSegmentIndex--];
1172                 if (seg.count != 0) {
1173                     currentTable = seg.table;
1174                     for (int j = currentTable.length - 1; j >= 0; --j) {
1175                         if ( (nextEntry = currentTable[j]) != null) {
1176                             nextTableIndex = j - 1;
1177                             return;
1178                         }
1179                     }
1180                 }
1181             }
1182         }
1183 
1184         public boolean hasNext() { 
1185             while (nextEntry != null) {
1186                 if (nextEntry.keyRef.get() != null) 
1187                     return true;
1188                 advance();
1189             }
1190             
1191             return false;
1192         }
1193 
1194         HashEntry<K,V> nextEntry() {
1195             do {
1196                 if (nextEntry == null)
1197                     throw new NoSuchElementException();
1198                 
1199                 lastReturned = nextEntry;
1200                 currentKey = lastReturned.keyRef.get();
1201                 advance();
1202             } while (currentKey == null); // Skip GC'd keys
1203             
1204             return lastReturned;
1205         }
1206 
1207         public void remove() {
1208             if (lastReturned == null)
1209                 throw new IllegalStateException();
1210             ConcurrentWeakHashMap.this.remove(currentKey);
1211             lastReturned = null;
1212         }
1213     }
