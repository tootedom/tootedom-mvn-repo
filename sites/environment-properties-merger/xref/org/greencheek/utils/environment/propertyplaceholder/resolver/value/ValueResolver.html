19  import java.util.Map;
20  import java.util.Properties;

27  public interface ValueResolver {
28  
29      /**
30       * Replaces all placeholders of format <code>${name}</code> with the corresponding property
31       * from the supplied {@link Properties}.  All entries in the properties map are visited on by one to
32       * resolve any placeholders from within the properties object.
33       *
34       * @param properties the <code>Properties</code> to use for replacement, and also the properties for which
35       *                   we loop around and resolve.
36       * @return the supplied value with placeholders replaced inline.
37       */
38      Properties resolvedPropertyValues(Properties properties,boolean trimValues);
39      Properties resolvedPropertyValues(Properties properties);
40  
41  
42      /**
43       * Replaces all placeholders of format <code>${name}</code> with the corresponding property
44       * from the supplied {@link Map<String,String>}.  All entries in the Map<String,String>
45       * map are visited on by one to resolve any placeholders from within the same Map object.
46       *
47       * @param properties the <code>Map<String,String></code> to use for replacement,
48       *                   and also the properties for which we loop around and resolve.
49       * @return the supplied value with placeholders replaced inline.
50       */
51      Map<String,String> resolvedPropertyValues(Map<String,String> properties,boolean trimValues);
52      Map<String,String> resolvedPropertyValues(Map<String,String> properties);
53      /**
54       * Replaces all placeholders of format <code>${name}</code> with the value returned from the supplied
55       * {@link Properties} object.
56       * @param key the entry within the given Properties object; for which we which to replace any placeholder that exist
57       *            in the value; from the set of other properties contained within the given Properties object.
58       * @param properties the <code>Properties</code> to use as the source of the placeholders replacements
59       * @return the supplied value with placeholders replaced inline.
60       */
61      String resolvedPropertyValue(Properties properties, String key,boolean trimValues);
62      String resolvedPropertyValue(Properties properties, String key);
63  
64      /**
65       * Replaces all placeholders of format <code>${name}</code> with the value returned from the supplied
66       * {@link Map} object.
67       * @param key the entry within the given Map object; for which we which to replace any placeholder that exist
68       *            in the value.  The replacement is taken from the set of other key/value pairs
69       *            also contained within the given Map object.
70       * @param map the <code>Map</code> to use as the source of the placeholders replacements
71       * @return the supplied value with placeholders replaced inline.
72       */
73      String resolvedPropertyValue(Map<String,String> map, String key,boolean trimValues);
74      String resolvedPropertyValue(Map<String,String> map, String key);
75  }
