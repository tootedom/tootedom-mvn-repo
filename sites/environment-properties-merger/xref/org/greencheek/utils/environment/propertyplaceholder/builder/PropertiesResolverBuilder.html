25  /**
26   * @{link #build} must be called on the same thread as the other setters.
27   * This is so that the @{link PropertiesResolver} is created with all the values that have been set on the builder
28   * object; otherwise the resulting resolver could be constructed with default values rather that those set on the
29   * builder object; due to the visibility rules of the jvm memory model
30   *
31   * User: dominictootell
32   * Date: 09/06/2012
33   * Time: 13:08
34   */
35  public interface PropertiesResolverBuilder {
36      final static boolean DEFAULT_TRIMMING_PROPERTY_VALUES = true;
37      final static ValueResolver DEFAULT_PROPERTY_VALUE_RESOLVER = new VariablePlaceholderValueResolver();
38  
39      public PropertiesResolver build(PropertiesMerger mergedPropertiesLoader);
40  
41      /**
42       * Returns a set of properties that are retrieved from the given properties merger
43       * with the values resolved of any variables they contained (if they can be resolved)
44       *
45       * @param mergedPropertiesLoader The merger that is responsible for reading the properties from the file system
46       * @return the set of the properties, from the merger, with the values resolved of any variables (if possible)
47       */
48      public Properties buildProperties(PropertiesMerger mergedPropertiesLoader);
49  
50      /**
51       * Returns a set of properties that are retrieved from the given properties merger (the merger is created
52       * from the PropertiesMergerBuilder) with the values resolved of any variables they contained (if they can be resolved)
53       *
54       * @param builder The builder object that can be used to create a merger
55       *                that is responsible for reading the properties from the file system; from which the properties
56       *                are obtained, and the values resolved.
57       *
58       * @return the set of the properties, from the merger, with the values resolved of any variables (if possible)
59       */
60      public Properties buildProperties(PropertiesMergerBuilder builder);
61  
62      public PropertiesResolverBuilder setPropertyValueResolver(ValueResolver resolver);
63      public ValueResolver getValueResolver();
64  
65      public PropertiesResolverBuilder setTrimmingPropertyValues(boolean b);
66      public boolean isTrimmingPropertyValues();
67  
68  
69  }
