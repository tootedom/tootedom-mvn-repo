16  package org.greencheek.utils.environment.propertyplaceholder.resolver;
17  
18  import org.greencheek.utils.environment.propertyplaceholder.builder.EnvironmentSpecificPropertiesMergerBuilder;
19  import org.greencheek.utils.environment.propertyplaceholder.builder.EnvironmentSpecificPropertiesResolverBuilder;
20  import org.greencheek.utils.environment.propertyplaceholder.merger.PropertiesMerger;
21  
22  import static org.junit.Assert.assertEquals;
23  
24  /**
25   * <p>
26   * Tests that we can create the resolver to merge the properties in the following order,
27   * Which are the defaults for the builders:
28   * <ol>
29   *     <li>/config/default.props</li>
30   *     <li>/config/environments/prod.props</li>
31   *     <li>/data/opsoverrides/appname/config/defaults.properties</li>
32   *     <li>/data/opsoverrides/appname/config/environments/prod.props</li>
33   * </ol>
34   * </p>
35   * <p>
36   *     The files are varied on the System property: ENV
37   * </p>
38   *
39   * This is not a unit test, it relies on /data/opsoverrides/appname/config, i want to actually test that
40   * the default read from the appropriate default locations.
41   *
42   * Looking for result of:
43   * <pre>
44   *     default=default
45         platform=prod
46         ops-override-env=prod
47         ops-override=default
48   * </pre>
49   * /data/opsoverrides/appname/config/environments/prod.props contains "ops-override-env=prod"
50   * /data/opsoverrides/appname/config/default.props contains "ops-override=default"
51   *
52   *
53   * User: dominictootell
54   * Date: 09/06/2012
55   * Time: 18:19
56   */
57  public class TestDefaultsForMergedPropertyResolving {
58  
59      static PropertiesResolver appNameResolver;
60      static PropertiesResolver nullAppNameResolver;
61  
62      public static void main(String[] args) {
63          setUp();
64          testPropertiesWithAppNameAreResolved();
65          testPropertiesWithoutAppNameAreResolved();
66      }
67  
68      public static void setUp() {
69          System.setProperty("ENV","prod");
70          PropertiesMerger merger = new EnvironmentSpecificPropertiesMergerBuilder().
71                                        setApplicationName("appname").build();
72  
73  
74          appNameResolver = new EnvironmentSpecificPropertiesResolverBuilder().build(merger);
75  
76          nullAppNameResolver = new EnvironmentSpecificPropertiesResolverBuilder().build(
77                  new EnvironmentSpecificPropertiesMergerBuilder().build()
78          );
79      }
80  
81  
82  
83      public static void testPropertiesWithAppNameAreResolved()
84      {
85          assertEquals("default",appNameResolver.getProperty("default"));
86          assertEquals("prod",appNameResolver.getProperty("platform"));
87          assertEquals("prod",appNameResolver.getProperty("ops-override-env"));
88          assertEquals("default",appNameResolver.getProperty("ops-override"));
89      }
90  
91      public static void testPropertiesWithoutAppNameAreResolved()
92      {
93          assertEquals("default",nullAppNameResolver.getProperty("default"));
94          assertEquals("prod",nullAppNameResolver.getProperty("platform"));
95          assertEquals("basedefault",nullAppNameResolver.getProperty("ops-override-env"));
96          assertEquals("basedefault",nullAppNameResolver.getProperty("ops-override"));
97      }
98  }
