16  package org.greencheek.utils.environment.propertyplaceholder.examples.spring.profile;
17  
18  import org.greencheek.utils.environment.propertyplaceholder.builder.EnvironmentSpecificPropertiesMergerBuilder;
19  import org.greencheek.utils.environment.propertyplaceholder.merger.PropertiesMerger;
20  import org.springframework.beans.factory.annotation.Autowired;
21  import org.springframework.beans.factory.annotation.Qualifier;
22  import org.springframework.context.ApplicationContextInitializer;
23  import org.springframework.context.ConfigurableApplicationContext;
24  import org.springframework.context.annotation.Bean;
25  import org.springframework.context.annotation.Configuration;
26  import org.springframework.context.support.PropertySourcesPlaceholderConfigurer;
27  import org.springframework.core.env.MutablePropertySources;
28  import org.springframework.core.env.PropertiesPropertySource;
29  
30  import java.util.Properties;
31  
32  /**
33   * User: dominictootell
34   * Date: 10/06/2012
35   * Time: 14:42
36   */
37  
38  @Configuration
39  public class AppBootstrap implements ApplicationContextInitializer {
40  
41  
42  
43      private static Properties environmentalProperties;
44      static {
45          PropertiesMerger merger = new EnvironmentSpecificPropertiesMergerBuilder().build();
46          environmentalProperties = merger.getMergedProperties();
47      }
48  
49      public static void initialise(ConfigurableApplicationContext applicationContext) {
50          applicationContext.getEnvironment().getPropertySources().addFirst(new PropertiesPropertySource("p",environmentalProperties));
51      }
52  
53  
54      @Autowired
55      private EnvBean envBean;
56  
57      @Bean
58      public static PropertySourcesPlaceholderConfigurer propertySourcesPlaceholderConfigurer() {
59          PropertySourcesPlaceholderConfigurer config = new PropertySourcesPlaceholderConfigurer();
60  
61  
62          MutablePropertySources sources = new MutablePropertySources();
63          sources.addFirst(new PropertiesPropertySource("p",environmentalProperties));
64  
65          config.setPropertySources(sources);
66  
67          return config;
68      }
69  
70      @Override
71      public void initialize(ConfigurableApplicationContext applicationContext) {
72          initialise(applicationContext);
73      }
74

77  
78  }
