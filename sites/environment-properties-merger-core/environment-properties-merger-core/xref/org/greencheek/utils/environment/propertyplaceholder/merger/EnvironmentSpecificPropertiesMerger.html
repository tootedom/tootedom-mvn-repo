35  /**
36   * Thread safe implementation of the @{link #PropertiesMerger}.  The implementation is constructed via a
37   * @{link #PropertiesMergerBuilder}.
38   *
39   * The properties are obtained first from either a location on the classpath or filesystem.  These set of
40   * properties can vary depending upon a given set of environment or system variables: i.e:
41   * <ol>
42   *     <li>Read default.props</li>
43   *     <li>Override the properties set in default.props with the contents
44   *     of a .properties file in the environments/ folder that represents the current value of the ENV system property, i.e. dev.properties</li>
45   * </ol>
46   * The properties files that are read that override default can vary not only on one ENV variable but could vary on several:
47   * <ol>
48   *     <li>Read default.props</li>
49   *     <li>Override the properties set in default.props with the contents
50   *     of a .properties file in the environments/ folder that represents the current value of the ${ENV} system property, i.e. dev.properties</li>
51   *     <li>Override dev.properties with a .properties file that represents the current value of ${ENV}.${OS.ARCH} system and env properties i.e. dev.x86_64.properties</li>
52   * </ol>
53   * The obtained properties' values are not modified in any way.  Variables contained within the values should not be
54   * modified or resolved.
55   *
56   * The implementation is thread safe.
57   *
58   * User: dominictootell
59   * Date: 09/06/2012
60   * Time: 12:26
61   */

93      public EnvironmentSpecificPropertiesMerger(PropertiesMergerBuilder builder) {
94          List<String> overrideFiles = new ArrayList<String>();
95          this.applicationName = builder.getApplicationName();
96  
97          ResourceLoader opsOverride =   builder.getResourceLoaderForOperationalOverrides();
98          if(opsOverride == PropertiesMergerBuilder.DEFAULT_RESOURCE_LOADER_FOR_OPERATION_OVERRIDES) {
99              if(applicationName!=null && applicationName.trim().length()>0) {
100                 opsOverride = new FileSystemResourceLoader(PropertiesMergerBuilder.DEFAULT_OPERATIONAL_OVERRIDE_LOCATION + applicationName + PropertiesMergerBuilder.DEFAULT_CONFIGURATION_LOCATION);
101             } else {
102                 // Make it null and do not read anything for the overrides
103                 opsOverride = null;
104             }
105         }
106 
107         this.operationalOverridesResourceLoader =  opsOverride;
108         this.resourceLoaderForLoadingConfigurationProperties = builder.getResourceLoaderForLoadingConfigurationProperties();
109 
110 
111         this.extensionForPropertiesFile = builder.getExtensionForPropertiesFile();
112         this.extensionSeparatorCharForPropertiesFile = builder.getExtensionSeparatorCharForPropertiesFile();
113 
114         this.nameOfDefaultPropertiesFile = builder.getNameOfDefaultPropertiesFile() +
115                 extensionSeparatorCharForPropertiesFile + extensionForPropertiesFile;
116 
117         this.strictMergingOfProperties = builder.isStrictMergingOfProperties();
118         this.delimiterUsedForSeparatingSwitchingConfigurationVariables = builder.getDelimiterUsedForSeparatingSwitchingConfigurationVariables();
119         this.operatingEnvironmentVariableReader = builder.getOperatingEnvironmentVariableReader();
120 
121         this.relativeLocationOfFilesOverridingDefaultProperties = builder.getRelativeLocationOfFilesOverridingDefaultProperties();
122         this.sensitivePropertyMasker = builder.getSensitivePropertyMasker();
123         this.outputtingPropertiesInDebugMode = builder.isOutputtingPropertiesInDebugMode();
124 
125 //        SERVER_ENV
126 //        SERVER_ENV.TARGET_ENV
127 //        SERVER_ENV.SERVER_TYPE
128 //        SERVER_ENV.SERVER_TYPE.TARGET_ENV
129 
130         String[][] configurationPossibilities = builder.getVariablesUsedForSwitchingConfiguration();
131 
132         for(int i = 0;i<configurationPossibilities.length;i++) {
133             String variables[] = configurationPossibilities[i];
134 
135             StringBuilder continuedFileName = (relativeLocationOfFilesOverridingDefaultProperties == null ||
136                     relativeLocationOfFilesOverridingDefaultProperties.trim().length()==0) ?
137                     new StringBuilder() :
138                     new StringBuilder(relativeLocationOfFilesOverridingDefaultProperties);
139             int noOfVars = 1;
140             boolean done = false;
141             for(String variable : variables) {
142 
143                 String variableValue = operatingEnvironmentVariableReader.getProperty(variable, operatingEnvironmentVariableReader.getEnv(variable, ""));
144                 if(variableValue==null || variableValue.trim().length()==0) break;
145                 else {
146                     continuedFileName.append(variableValue);
147                     if(noOfVars++!=variables.length) continuedFileName.append(delimiterUsedForSeparatingSwitchingConfigurationVariables);
148                     else done = true;
149                 }
150             }
151 
152             if(done) {
153                 continuedFileName.append(extensionSeparatorCharForPropertiesFile).append(extensionForPropertiesFile);
154                 overrideFiles.add(continuedFileName.toString());
155             }
156 
157         }
158 
159         possibleOverrideFiles = overrideFiles;
160         mergedProperties = mergeProperties();
161 
162     }

164     /**
165      * <p>
166      * uses java.util.Properties.load(InputStream), to load in the given stream into a
167      * Properties object.  This method also checks to see if
168      * the logger is in debug mode and output the contents of the properties object into it.
169      * </p>
170      * <p>
171      * Developers need to be aware that this means  sensitive information could leak into log files
172      * so they need to ensure that they aren't logging at debug level on live platforms.
173      * </p>
174      * <p>
175      * A Pattern masking object can be used to blank out sensitive properties, so that output does
176      * not leak into the log files
177      * </p>
178      *
179      * @param stream
180      * @return
181      * @throws java.io.IOException
182      */
183     private Properties load(InputStream stream) throws IOException {
184 
185         Properties properties = new Properties();
186         properties.load(stream);
187 
188         if (log.isDebugEnabled() && isOutputtingPropertiesInDebugMode()) {
189 
190             for (Object keyItem : properties.keySet()) {
191                 String key = (String)keyItem;
192                 String value = properties.getProperty(key);
193 
194                 Matcher m = getSensitivePropertyMasker().matcher(key);
195                 if(m.matches()) {
196                     log.debug(key + " : " + "######");
197                 }
198                 else {
199                     log.debug(key + " : " + value);
200                 }
201 
202             }
203         }
204         return properties;
205     }

207     /**
208      * For a given file, that represents a properties file, loads the contents into a Properties file
209      * @param resource the file to convert to a Properties file.
210      * @return a Properties object, will always return a properties object.
211      */
212     private Properties load(Resource resource) {
213         if(resource == null || !resource.isAvailable()) {
214             return new Properties();
215         }
216 
217         InputStream is = null;
218         Properties p = null;
219         try {
220             log.debug("loading properties from {}",resource);
221             is = resource.getStream();
222             p = load(is);
223         } catch(IOException e ) {
224             log.warn("Unable to read file: {}",resource,e);
225         } finally {
226             if(is!=null) {
227                 try {
228                     is.close();
229                 } catch (Exception e) {
230                     log.warn("Unable to close file:{}",resource,e);
231                 }
232             }
233         }
234 
235         return (p==null) ? new Properties() : p;
236     }

238     /**
239      * Merges two properties objects by creating a new Properties object, adding all of the
240      * first set of properties and then adding properties from the second. In effect, the first
241      * set of properties is the default.<br/>
242      * If the merging is strict, then an exception is throw if a property exists
243      * in the second properties object that does not exist in the first.
244      *
245      *
246      *
247      * @param first can be null, the first set of properties that represents the defaults to be overridden
248      * @param second can be null, the second set of properties that override the defaults in the first set of properties
249      * @return The merge of the second properties over the top of the first set of properties
250      */
251     public static Properties mergeProperties(Properties first, Properties second, boolean strict) throws NoMatchingPropertyException {
252 
253         final Properties merged = new Properties();
254 
255 
256         if(first!=null){
257             for (final Object entry : first.keySet()) {
258                 final String key = entry.toString();
259                 merged.setProperty(key,first.getProperty(key));
260             }
261         }
262 
263         if(second!=null) {
264             for (Object key : second.keySet()) {
265                 if (merged.get(key)==null) {
266                     String msg = "NoMatchingPropertyWarning: Property \"" + key + "\" from overriding properties does not exist in original properties";
267                     if (strict) {
268                         log.warn(msg);
269                         throw new NoMatchingPropertyException(msg);
270                     } else {
271                         log.warn(msg);
272                     }
273                 }
274             }
275         }
276 
277         if (second != null) {
278             for (final Object entry : second.keySet()) {
279                 final String key = entry.toString();
280                 merged.setProperty(key,second.getProperty(key));
281             }
282         }
283 
284         return merged;
285 
286     }

288     private Resource[] loadFiles(List<String> files, ResourceLoader resourceLoader) {
289         List<Resource> foundFiles = new ArrayList<Resource>(files.size());
290 
291         for(String location : files) {
292             log.debug("Checking for override file: {} in location {}",location,resourceLoader.getBaseLocation());
293             Resource override = resourceLoader.getFile(location);
294             if(override.isAvailable()) {
295                 log.debug("Override {} is available for sourcing",override);
296                 foundFiles.add(override);
297             }
298         }
299 
300         return foundFiles.toArray(new Resource[foundFiles.size()]);
301     }

303     private Properties mergeProperties() {
304         Properties merged;
305 
306         ResourceLoader defaultAppPropertiesLoader = getResourceLoaderForLoadingConfigurationProperties();
307         log.debug("Attempting to source default properties file {} from location {}",getNameOfDefaultPropertiesFile(),defaultAppPropertiesLoader.getBaseLocation());
308 
309         Resource defaultProperties = defaultAppPropertiesLoader.getFile(getNameOfDefaultPropertiesFile());
310         if(!defaultProperties.isAvailable()) {// || !defaultProperties.canRead()) {
311             String message = "Unable to source default properties file:" + getNameOfDefaultPropertiesFile();
312             log.warn(message);
313             if(isStrictMergingOfProperties())
314                 throw new NoDefaultPropertiesFileException(message);
315         }
316 
317 
318         merged = load(defaultProperties);
319 
320         for(Resource file : loadFiles(getPossibleOverrideFiles(),defaultAppPropertiesLoader)) {
321             merged = mergeProperties(merged,load(file),strictMergingOfProperties);
322 
323         }
324 
325         // read overrides
326         ResourceLoader overridesResourceLoader = getOperationalOverridesResourceLoader();
327         if(overridesResourceLoader!=null)
328             merged = mergePropertiesAgainstOverrides(merged,overridesResourceLoader);
329 
330         return merged;
331     }

333     /**
334      * This method provided the functionality of overriding the currently read properties; against
335      * those in a specific location that can be override by a operations team; without the need to modify the
336      * application code.
337      *
338      * @param currentProperties The current properties that are to be override from the set of overrides
339      *                          that are to be found by the overrides location's resource loader
340      */
341     protected Properties mergePropertiesAgainstOverrides(Properties currentProperties, ResourceLoader overridesLocationLoader) {
342         Properties merged = new Properties();
343 
344         if(currentProperties!=null){
345             for (final Object entry : currentProperties.keySet()) {
346                 final String key = entry.toString();
347                 merged.setProperty(key,currentProperties.getProperty(key));
348             }
349         }
350 
351         List<String> possibleOverrideFiles = new ArrayList<String>(getPossibleOverrideFiles());
352         possibleOverrideFiles.add(0,getNameOfDefaultPropertiesFile());
353 
354         for(Resource file : loadFiles(possibleOverrideFiles,overridesLocationLoader)) {
355             merged = mergeProperties(merged,load(file),strictMergingOfProperties);
356         }
357 
358         return merged;
359     }
