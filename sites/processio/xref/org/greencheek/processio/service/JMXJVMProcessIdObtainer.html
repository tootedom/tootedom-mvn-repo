16  package org.greencheek.processio.service;
17  
18  import java.lang.management.ManagementFactory;
19  import java.lang.management.RuntimeMXBean;
20  
21  /**
22   * Used JMX to obtain the PID of the JVM process.
23   * Parsing the Name attribute of the java.lang:type=Runtime JMX Mbean.
24   *
25   * User: dominictootell
26   * Date: 22/04/2012
27   * Time: 12:59
28   */
29  public class JMXJVMProcessIdObtainer implements JVMProcessIdObtainer {
30  
31      private static final int CURRENT_JVM_PID;
32      static {
33          CURRENT_JVM_PID = parsePid(ManagementFactory.getRuntimeMXBean());
34      }
35  
36      @Override
37      public int getProcessId() {
38          return CURRENT_JVM_PID;  //To change body of implemented methods use File | Settings | File Templates.
39      }
40  
41      private static int parsePid(RuntimeMXBean rtbean) {
42          String s = rtbean.getName();
43          int index = s.indexOf('@');
44          String pid;
45          if (index != -1) {
46              pid = s.substring(0, index);
47              try
48              {
49                  return Integer.parseInt(pid);
50              }
51              catch(NumberFormatException e){
52                  return PID_NOT_FOUND;
53              }
54          } else {
55              return PID_NOT_FOUND;
56          }
57      }
58  }
